```typescript
export const HOT_STAGES = ["qualified", "contacted", "demo", "proposal", "negotiation"] as const;
export type RuleSeverity = "critical" | "warning" | "info";

export interface RuleInsight {
  rule_id: string;       // "R1" | "R2" | "R3" | "R4"
  severity: RuleSeverity;
  evidence: string;      // human-readable explanation
  module_tags: string[]; // e.g. ["sales", "finance"]
  generated_at: string;  // ISO date string
}

export interface SalesSnapshot {
  leads: Lead[];
  recent_activity: LeadActivity[];
}
export interface HireSnapshot {
  candidates: CandidateWithEvaluation[];
  roles: Role[];
}
export interface FinanceSnapshot {
  invoices: Invoice[];
  expenses: Expense[];
  runway: RunwayData | null;
}
export interface CrossModuleSnapshot {
  sales: SalesSnapshot | null;
  hire: HireSnapshot | null;
  finance: FinanceSnapshot | null;
  errors: string[];
  generated_at: string;
}

export function isHotLead(lead: Lead): boolean;
// Returns true for status: qualified, contacted, demo, proposal, negotiation
// Returns false for: new, won, lost, nurture
```

RunwayData conversion:
- `runway_months * 30` = days of runway (R1 threshold: < 60 days, R2 threshold: < 90 days)
- InvoiceStatus: "odendi" = paid, "beklemede" = pending, "gecmis" = overdue

  <behavior>
    R1 — checkR1RunwayHotLeads(snapshot: CrossModuleSnapshot): RuleInsight | null
    - snapshot.finance null → null
    - snapshot.sales null → null
    - finance.runway null → null
    - runway_months * 30 >= 60 → null (healthy runway)
    - runway_months * 30 < 60 AND no hot leads → null
    - runway_months * 30 < 60 AND 1+ hot lead → RuleInsight {
        rule_id: "R1", severity: "critical",
        evidence: "{N} hot lead(s) in pipeline, runway at {days} days",
        module_tags: ["sales", "finance"]
      }

    R2 — checkR2HireRunway(snapshot: CrossModuleSnapshot): RuleInsight | null
    - snapshot.finance null → null
    - snapshot.hire null → null
    - finance.runway null → null
    - runway_months * 30 >= 90 → null
    - runway_months * 30 < 90 AND no active roles (role.active=true) → null
    - runway_months * 30 < 90 AND 1+ active role → RuleInsight {
        rule_id: "R2", severity: "warning",
        evidence: "{N} open role(s) hiring, runway at {days} days",
        module_tags: ["hire", "finance"]
      }

    R3 — checkR3DealsCandidates(snapshot: CrossModuleSnapshot): RuleInsight | null
    - snapshot.sales null → null
    - snapshot.hire null → null
    - No leads with status "won" in the last 7 days (check updated_at) → null
    - No candidates with status "reviewed" in the last 7 days (check applied_at) → null
    - Both a won lead AND a reviewed candidate exist within last 7 days → RuleInsight {
        rule_id: "R3", severity: "info",
        evidence: "Deal closing and candidate advancing simultaneously — consider capacity timing",
        module_tags: ["sales", "hire"]
      }
    - "Last 7 days" = Date.now() - 7 * 24 * 60 * 60 * 1000

    R4 — checkR4RevenueSilence(snapshot: CrossModuleSnapshot): RuleInsight | null
    - snapshot.finance null → null
    - snapshot.sales null → null
    - Invoice created within last 14 days (check created_at) → null (not silent)
    - Lead with status "won" exists (any time) → null (won signal present)
    - No invoice in 14 days AND no won leads → RuleInsight {
        rule_id: "R4", severity: "warning",
        evidence: "No invoices sent in {N} days and no deals won — revenue pipeline may be stalling",
        module_tags: ["finance", "sales"]
      }
    - Days since last invoice: compute from most recent invoice created_at or use 14+ if none exist
  </behavior>
  <implementation>
    Each rule file follows the same structure:
    1. Import CrossModuleSnapshot, RuleInsight from lib/intelligence/types.ts
    2. Import isHotLead from lib/intelligence/types.ts (R1 only)
    3. Export a single named function (e.g., checkR1RunwayHotLeads)
    4. Start with null-guard on required modules: `if (!snapshot.finance || !snapshot.sales) return null`
    5. Compute condition, return RuleInsight or null

    For R1 and R2: `const runwayDays = snapshot.finance.runway.runway_months * 30`
    For R3: Use `new Date(lead.updated_at).getTime()` for won leads, `new Date(candidate.applied_at).getTime()` for reviewed candidates.
      NOTE: Candidate has no updated_at or reviewed_at field — applied_at is used as a proxy for
      "candidate advancing recently". This means R3 fires based on when a candidate applied, not
      when they reached "reviewed" status. This is an accepted v1 limitation (per 03-CONTEXT.md
      Claude's Discretion). Document the proxy in an implementation comment:
      // applied_at used as proxy for "candidate advancing recently" — Candidate type has no
      // reviewed_at or updated_at field. v1 limitation: R3 may fire for old applications
      // that were reviewed late rather than truly recent candidate advancement.
    For R4: Find most recent invoice by sorting on created_at, compute days since last invoice

    Severity assignments (use judgment per context):
    - R1: "critical" (runway + hot leads = cash-flow emergency signal)
    - R2: "warning" (hiring costs runway not yet critical)
    - R3: "info" (timing heads-up, not an emergency)
    - R4: "warning" (revenue silence is concerning but not critical)
